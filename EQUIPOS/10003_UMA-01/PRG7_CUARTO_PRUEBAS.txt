REM SINCRONIZACION CON CONTROLADOR CUARTO DE PRUEBAS

	REM *VARIABLE DE ACTIVACIÃ“N

		IF INTERVAL(00:00:02) THEN
			BV32 = 10021.BV1
			IF BV32 = 1 THEN 10021.BV2 = BV3 ELSE 10021.BV2 = 0
		ENDIF

	REM *BLOQUEO DE MODO DE ARRANQUE

		IF BV32 = 1 THEN
			MSV1@7 = 4
		 ELSE
			RLQ MSV1@7
		ENDIF

END

REM PRG5_UMA (CONTROL: CUARTO CONTROL 1)
REM CONTROL DE UMA
REM AUTHOR CARLOS JIMENEZ HIRASHI @cjhirashi
REM VERSION 2.0.0

REM DECLARACION DE VARIABLES DEL SISTEMA
	LOCALS SS_CP, ST_UMA

REM LOGICA DE CONTROL

	REM SINCRONIZACION DE DATOS
	IF INTERVAL(00:00:05) THEN
		BV2 = 10003.BV3

		10003.AV4 = AV118

		10003.AV8 = AV119

		IF BV1 = 1 THEN
			10003.MSV1@7 = 4
		 ELSE
			RLQ 10003.MSV1@7
		ENDIF
	ENDIF

END
